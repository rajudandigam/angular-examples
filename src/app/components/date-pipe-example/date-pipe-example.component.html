Current date with date pipe

input: currentDate | date

{{ currentDate | date }}

Current date with date pipe and short format

input: currentDate | date: 'short'

{{ currentDate | date: 'short'}}

Current date with date pipe and mediumTime format

input: currentDate | date: 'mediumTime'

{{ currentDate | date: 'mediumTime'}}

Current date with date pipe and custom format

input: currentDate | date: 'yyyy-MM-dd HH:mm:ss'

{{ currentDate | date: 'yyyy-MM-dd HH:mm:ss'}}

Current date with date pipe and medium format and time zone

input: currentDate | date: 'medium':'IST'

{{ currentDate | date: 'medium':'IST' }}

Current date with date pipe and medium format and time zone and locale

input: currentDate | date: 'medium':'GMT':'en-GB'

{{ currentDate | date: 'medium':'GMT':'en-GB' }}
